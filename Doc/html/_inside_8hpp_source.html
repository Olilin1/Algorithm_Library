41/*
42Function to determine if a point is inside, outside, or on the edge of a simple polygon.
43Takes as input a vector containing the points of the simple polygon and a point p.
44Returns a Status enum.
45If the point is inside the polygon Status::Inside is returned.
46If the point is outside the polygon Status::Outside is returned.
47If the point is on the edge of the polygon Status::OnEdge is returned.
48Time complexity: O(N)
49*/
50template <typename T>
51Status inside_poly(std::vector<Point<T>> const & polygon, Point<T> const & p){
52 double angleSum = 0;
53 size_t n = polygon.size();
54 for(size_t i = 0; i < n; i++){
55 if(onSegment({polygon[i], polygon[(i+1)%n]}, p)){
56 std::cerr << i << ' ' << ((i+1)%n) << '\n';
57 return OnEdge;
58 }
59 angleSum += angle(subtract(polygon[i], p), subtract(polygon[(i+1)%n], p));
60 }
61
62
63 if(abs(angleSum) < 0.00001) return Outside;
64 if(abs(abs(angleSum) - 2.0*(3.141592)) < 00001) return Inside;
65
66 throw std::logic_error("Can't determine point position");
67}
