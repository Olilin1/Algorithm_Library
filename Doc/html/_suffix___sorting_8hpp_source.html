59SuffixArray::SuffixArray(std::string const & s){
60 std::vector<size_t> buckets(s.length());
61 std::vector<size_t> newBuckets(s.length());
62 std::vector<size_t> suffixPosition(s.length());
63
64 buckets[0] = 0;
65 newBuckets[0] = 0;
66
67 for(size_t i = 0; i < s.size(); i++){
68 sorted.push_back(i);
69 }
70
71 //Do an initial sorting on just the first charachter
72 std::sort(sorted.begin(), sorted.end(), [&s](const size_t a, const size_t b){
73 return s[a] < s[b];
74 });
75
76 buckets[0] = 0;
77 suffixPosition[sorted[0]] = 0;
78
79 for(int i = 1; i < s.length(); i++){
80 if(s[sorted[i]] == s[sorted[i-1]]) buckets[i] = buckets[i-1];
81 else buckets[i] = buckets[i-1] + 1;
82
83 suffixPosition[sorted[i]] = buckets[i];
84 }
85
86
87
88 //This loop will run log(n) times
89 for(int k = 1; k <= s.length(); k *= 2){
90
91 //Sorting takes n log(n) time
92 std::sort(sorted.begin(), sorted.end(), [&s, &k, &suffixPosition](const size_t a, const size_t b){
93 if(suffixPosition[a] == suffixPosition[b]){
94 //If one of the suffixes in this buckets end then it should come before the others
95 if(a+k >= s.length() || b+k >= s.length()){
96 return a > b; // '>' instead of '<' because the bigger the number, the shorter the suffix
97 }
98 //Otherwise we sort based on the next 2^k chars, which have already been sorted somewhere else.
99 return suffixPosition[a+k] < suffixPosition[b+k];
100 }
101 //If they don't belong to the same bucket then their relative position shall remain the same.
102 return suffixPosition[a] < suffixPosition[b];
103 });
104
105 //This will run n times for bookkeeping
106 for(int i = 1; i < s.length(); i++){
107 if(buckets[i] > buckets[i-1]) newBuckets[i] = newBuckets[i-1] + 1;
108 else if(suffixPosition[sorted[i] + k] > suffixPosition[sorted[i-1] + k]) newBuckets[i] = newBuckets[i-1] + 1;
109 else newBuckets[i] = newBuckets[i-1];
110 }
111
112 buckets.swap(newBuckets);
113
114 for(int i = 0; i < s.length(); i++){
115 suffixPosition[sorted[i]] = buckets[i];
116 }
117
118 }
119}
