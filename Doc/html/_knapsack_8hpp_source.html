12A function that solves the knapsack problem and returns the indices in the items vector that should be used to obtain the optimal value.

20std::vector<size_t> knapsack(const int capacity, const std::vector<std::pair<int, int>>& items){

23 std::vector<std::vector<int>> dp(capacity+1);
24 for(size_t i = 0; i <= capacity; i++){
25 dp[i].resize(items.size()+1);

28 //Fill out DP
29 for(int weight = capacity; weight >= 0; --weight){
30 for(int item = items.size(); item >= 0; --item){
31 if(weight == capacity || item == items.size())
32 dp[weight][item] = 0;
33 else if (weight+items[item].first <= capacity)
34 dp[weight][item] = std::max(dp[weight][item + 1], dp[weight+items[item].first][item + 1] + items[item].second);
35 else
36 dp[weight][item] = dp[weight][item + 1];

38 }
