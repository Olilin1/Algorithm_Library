11/*
12Function that takes a vector of points and returns the convex hull encircling them.
13The hull is given as a list of points in counterclockwise order, each point being a node in the hull.
14If the points are all colinear then the two endpoints are given.
15
16Time Complexity: O(N log(n))
17*/
18
19template<typename T>
20std::vector<Point<T>> convex_hull(std::vector<Point<T>> points){
21 int leftMostPoint = 0;
22 Point<T> org = {0,0};
23 std::vector<Point<T>> hull;
24
25 //This is to later remove duplicates
26 std::sort(points.begin(), points.end(), [&org](Point<T> const & a, Point<T> const & b){
27 if(a.x == b.x)
28 return a.y < b.y;
29 else
30 return a.x < b.x;
31 });
32
33 //remove duplicates
34 std::vector<Point<T>> newPoints;
35 newPoints.push_back(points[0]);
36 for(int i = 1; i < points.size(); i++){
37 if(points[i] != *(newPoints.rbegin())) newPoints.push_back(points[i]);
38 }
39
40
41 points.swap(newPoints);
42
43 //Find the leftmost point that is definitely on the hull
44 for(int i = 1; i < points.size(); i++){
45 if(points[i].x < points[leftMostPoint].x) leftMostPoint = i;
46 else if(points[i].x == points[leftMostPoint].x && points[i].y > points[leftMostPoint].y) leftMostPoint = i;
47 }
48
49 org = points[leftMostPoint];
50 hull.push_back(points[leftMostPoint]);
51 points.erase(points.begin() + leftMostPoint);
52
53 //Sort the points based on angle from the previously determined point
54 std::sort(points.begin(), points.end(), [&org](Point<T> const & a, Point<T> const & b){
55 T cross = cross_product(subtract(a, org), subtract(b, org));
56 if(cross == 0) return distSquared(org, a) < distSquared(org, b); //De är ko-linjära
57 else return cross > 0;
58 });
59
60 //add points to the hull
61 for(Point<T> & p : points){
62
63 //if we are doing a rightturn pop last point of hull.
64 while(hull.size() >= 2 && cross_product(subtract(hull[hull.size()-1], hull[hull.size()-2]), subtract(p, hull[hull.size()-2])) <= 0) hull.pop_back();
65 hull.push_back(p);
66 }
67 return hull;
68}
