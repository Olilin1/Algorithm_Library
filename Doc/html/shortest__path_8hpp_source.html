14/*
15My implementation of the dijkstra algorithm.
16Returns a vector of pairs where the second item of each element i is the distance to the node i and the first item is the node you go to i from.
17If there is no way to reach a node i then v[i].first and v[i].second is set to the max value of size_t and unsigned long long int respectively.
18The start node's parent is set to the maximum value of size_t
19Takes as input the graph represented in adjacency list form and the node we start from.
20Time complexity: O(m * log(n)) where n is the amount of nodes in the graph and m is the amount of edges.
21*/
22
23typedef unsigned long long int ull;
24#define node first
25#define parent first
26#define distance second
27#define weight second

29std::vector<std::pair<size_t, ull>> dijkstra_shortest_path(const std::vector<std::vector<std::pair<size_t, ull>>>& adjaceny_list, size_t startNode){
30 struct Compare{
31 bool operator() (std::pair<size_t, ull> a, std::pair<size_t, ull> b){
32 return a.distance > b.distance;
33 }
34 };
35
36 std::vector<std::pair<size_t, ull>> parent_dist_pair(adjaceny_list.size(), {std::numeric_limits<size_t>::max(), std::numeric_limits<ull>::max()});
37 std::priority_queue<std::pair<size_t, ull>, std::vector<std::pair<size_t, ull>>, Compare> pq;
38
39 pq.push({startNode, 0});
40 parent_dist_pair[startNode].distance = 0;
41
42 while(!pq.empty()){
43 std::pair<size_t, ull> p = pq.top();
44 pq.pop();
45
46 if(p.distance != parent_dist_pair[p.node].distance)
47 continue;
48
49 for(auto edge : adjaceny_list[p.node]){
50 if(parent_dist_pair[edge.node].distance > p.distance + edge.weight) {
51 pq.push({edge.node, p.distance + edge.weight});
52 parent_dist_pair[edge.node].distance = p.distance + edge.weight;
53 parent_dist_pair[edge.node].parent = p.node;
54 }
55 }
56 }
57
58 return parent_dist_pair;
59}
