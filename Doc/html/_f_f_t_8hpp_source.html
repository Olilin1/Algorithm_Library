74std::vector<ll> Polymul(std::vector<ll> a, std::vector<ll> b){
75 int n=1;
76 int org = a.size() + b.size() - 1;
77
78 while(n < org) n*=2;
79
80 while(a.size() < n) a.push_back(0);
81 while(b.size() < n) b.push_back(0);
82
83 std::vector<std::complex<double>> a_value_rep = FFT(a, n);
84 std::vector<std::complex<double>> b_value_rep = FFT(b, n);
85 std::vector<std::complex<double>> merged;
86 std::vector<std::complex<double>> reversed;
87
88 merged.reserve(n);
89
90 for(int i = 0; i < n; i++){
91 merged.push_back(a_value_rep[i] * b_value_rep[i]);
92 }
93
94 reversed = INVERSE_FFT(merged, n);
95
96 std::vector<ll> ans;
97 ans.reserve(org);
98
99 for(int i = 0; i < org; i++){
100 ans.push_back(round(reversed[i].real()/(double)n));
101 }
102
103 return ans;
104}

107std::vector<std::complex<double>> FFT(const std::vector<ll>& p, int n, int start, int skip){
108 using namespace std::complex_literals;
109
110 if(p.size() == skip){
111 return {{(double)p[start], 0.0}};
112 } //Ett polynom av grad 0 har alltid värdet av koefficienten. FIIIIIIIX
113
114
115
116 std::vector<std::complex<double>> y_even, y_odd;
117 y_even = FFT(p, n/2, start, skip*2);
118 y_odd = FFT(p, n/2, start+skip, skip*2);
119
120 std::vector<std::complex<double>> y_merged(n);
121 for(int i = 0; i < n/2; i++){
122 std::complex<double> compute = std::exp((2.0*PI*1i*(double)i)/(double)n) * y_odd[i];
123 y_merged[i] = y_even[i] + compute;
124 y_merged[i+n/2] = y_even[i] - compute;
125 }
126 return y_merged;
127}

130std::vector<std::complex<double>> INVERSE_FFT(const std::vector<std::complex<double>>& p, int n, int start, int skip){
131 using namespace std::complex_literals;
132
133 if(p.size() == skip){
134 return {p[start]};
135 } //Ett polynom av grad 0 har alltid värdet av koefficienten. FIIIIIIIX
136
137 std::vector<std::complex<double>> y_even, y_odd;
138 y_even = INVERSE_FFT(p, n/2, start, skip*2);
139 y_odd = INVERSE_FFT(p, n/2, start+skip, skip*2);
140
141 std::vector<std::complex<double>> y_merged(n);
142 for(int i = 0; i < n/2; i++){
143 std::complex<double> compute = std::exp((-2.0*PI*1i*(double)i)/(double)n) * y_odd[i];
144 y_merged[i] = y_even[i] + compute;
145 y_merged[i+n/2] = y_even[i] - compute;
146 }
147 return y_merged;

150std::vector<ll> Polymul_Quadratic(std::vector<ll> a, std::vector<ll> b){
151 std::vector<ll> res(a.size() + b.size() - 1);
152
153 for(int i = 0; i < a.size(); i++){
154 for(int j = 0; j < b.size(); j++){
155 res[i+j] += a[i]*b[j];
156 }
157 }
